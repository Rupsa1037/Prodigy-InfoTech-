import matplotlib.pyplot as plt
import numpy as np

# Group the data by 'game' and 'sentiment', and count the occurrences
grouped_data = data.groupby(['game', 'sentiment']).size().unstack()

# Get the categories and sentiments
games = grouped_data.index
sentiments = grouped_data.columns

# Set up the plot
fig, ax = plt.subplots(figsize=(14, 7))

# Plot each sentiment category for each game
bar_width = 0.2
index = np.arange(len(games))

for i, sentiment in enumerate(sentiments):
    ax.bar(index + (i * bar_width), grouped_data[sentiment], width=bar_width, label=sentiment)

# Add labels and title
ax.set_xlabel('Game')
ax.set_ylabel('Count')
ax.set_title('Counts of Sentiments by Game')
ax.set_xticks(index + (bar_width * len(sentiments)) / 2)
ax.set_xticklabels(games)
ax.legend(title='Sentiment')

plt.show()
